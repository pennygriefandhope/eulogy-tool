  <script>
    // Initialize Lucide Icons
    lucide.createIcons();

    // The logic to handle the steps
    let currentStep = 1;
    const totalSteps = 4;
    
    const sections = document.querySelectorAll('.step-section');
    const btnNext = document.getElementById('btn-next');
    const btnBack = document.getElementById('btn-back');
    const stepIndicator = document.getElementById('step-indicator');
    const footerNote = document.getElementById('footer-note');

    // The Ethical Edit Options from your React Code
    const editOptions = [
      { 
        title: 'Struggle with mental health', 
        included: "They had depths I didn't fully understand.", 
        excluded: "Don't mention specific diagnoses or private suffering." 
      },
      { 
        title: 'Addiction or substance use', 
        included: "They were fighting something hard that I wish I could have fixed.", 
        excluded: "Don't list specifics or suggest they caused their death." 
      },
      { 
        title: 'Estrangement from family', 
        included: "They had relationships that mattered deeply to them, some visible, some private.", 
        excluded: "Don't explain the family conflict in detail." 
      },
      { 
        title: 'LGBTQ+ identity (if not publicly known)', 
        included: "They loved deeply and authentically.", 
        excluded: "Don't out someone if the family isn't ready." 
      },
      { 
        title: 'Financial struggles', 
        included: "They worked hard in ways most people never knew.", 
        excluded: "Don't air specific financial details." 
      }
    ];

    // Build the Ethical Edits step dynamically
    const editsContainer = document.getElementById('ethical-edits-container');
    editOptions.forEach((option, index) => {
      const card = document.createElement('div');
      card.className = "p-4 border border-stone-200 rounded-lg hover:border-stone-400 transition-colors cursor-pointer bg-white group";
      card.innerHTML = `
        <div class="flex items-center justify-between mb-2">
          <h3 class="font-bold text-stone-900 group-hover:text-stone-700">${option.title}</h3>
          <input type="checkbox" class="w-5 h-5 accent-stone-800 cursor-pointer">
        </div>
        <div class="text-sm space-y-2 mt-3 pl-2 border-l-2 border-stone-200 text-stone-600">
          <p><span class="font-bold text-emerald-700">Instead of:</span> ${option.excluded}</p>
          <p><span class="font-bold text-stone-800">Try saying:</span> "${option.included}"</p>
        </div>
      `;
      // Allow clicking anywhere on the card to toggle the checkbox
      card.addEventListener('click', (e) => {
        if(e.target.tagName !== 'INPUT') {
          const checkbox = card.querySelector('input');
          checkbox.checked = !checkbox.checked;
        }
      });
      editsContainer.appendChild(card);
    });

    // Navigation Logic
    function updateUI() {
      // Show correct section
      sections.forEach((sec, idx) => {
        if (idx + 1 === currentStep) {
          sec.classList.add('active');
        } else {
          sec.classList.remove('active');
        }
      });

      // Update Header Text
      stepIndicator.textContent = `Step ${currentStep} of ${totalSteps}`;

      // Update Buttons
      btnBack.style.visibility = currentStep === 1 ? 'hidden' : 'visible';
      
      if (currentStep === totalSteps) {
        btnNext.style.display = 'none';
        stepIndicator.style.display = 'none';
        footerNote.style.display = 'none';
      } else {
        btnNext.style.display = 'flex';
        stepIndicator.style.display = 'block';
        footerNote.style.display = 'block';
      }
    }

    btnNext.addEventListener('click', () => {
      if (currentStep < totalSteps) {
        currentStep++;
        updateUI();
      }
    });

    btnBack.addEventListener('click', () => {
      if (currentStep > 1) {
        currentStep--;
        updateUI();
      }
    });
  </script>
